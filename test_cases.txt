# Expert System Test Cases Catalog
# This file is for visualization/documentation of the tested scenarios.
# It mirrors test_suite.py in a readable, input.txt-like format.
#
# Conventions:
# - Rules use => and <=>
# - Initial facts line starts with '='
# - Query line starts with '?'
# - Expected values are documented in comments under each case
# - Default mode in this project: symbols start FALSE unless proven otherwise
# - UNDETERMINED appears only in ambiguous OR/XOR conclusion situations

################################################################################
# CASE 01: Direct implication
################################################################################
A => B
=A
?B
# Expected:
# B = TRUE

################################################################################
# CASE 02: Default false without support
################################################################################
A => B
=A
?D
# Expected:
# D = FALSE

################################################################################
# CASE 03: AND on LHS
################################################################################
A + B => C
=AB
?C
# Expected:
# C = TRUE

################################################################################
# CASE 04: NOT on LHS
################################################################################
A + !B => F
=A
?F
# Expected:
# F = TRUE

################################################################################
# CASE 05: AND on RHS sets both symbols
################################################################################
A => B + C
=A
?BC
# Expected:
# B = TRUE
# C = TRUE

################################################################################
# CASE 06: Biconditional expands both directions
################################################################################
A <=> B
=A
?B
# Expected:
# B = TRUE

################################################################################
# CASE 07: OR in conclusion becomes UNDETERMINED when active
################################################################################
A => B | C
=A
?BC

# B = UNDETERMINED
# C = UNDETERMINED

# CASE 08: OR in conclusion inactive keeps default FALSE
################################################################################
A => B | C
=
?BC

# B = FALSE
# C = FALSE

################################################################################
# CASE 09: XOR in conclusion becomes UNDETERMINED when active
################################################################################

=A
?BC
# Expected:
# B = UNDETERMINED
# C = UNDETERMINED

################################################################################
# CASE 10: Negated conclusion sets symbol FALSE
################################################################################
A => !V
=A
?V
# Expected:
# V = FALSE
